# NFL Betting System Makefile

.PHONY: help install dev-setup verify download train backtest dashboard test clean

help:
	@echo "NFL Betting System - Available Commands:"
	@echo "  make install       - Install dependencies"
	@echo "  make dev-setup     - Install dev dependencies"
	@echo "  make verify        - Verify setup"
	@echo "  make download      - Download NFL data"
	@echo "  make train         - Train models"
	@echo "  make backtest      - Run backtest"
	@echo "  make dashboard     - Launch dashboard"
	@echo "  make test          - Run tests"
	@echo "  make clean         - Clean generated files"

install:
	pip install -r requirements.txt

dev-setup:
	pip install -r requirements.txt
	pip install pytest pytest-cov black flake8
	pre-commit install

verify:
	python scripts/verify_setup.py

download:
	python scripts/download_data.py

train:
	python scripts/train_model.py

backtest:
	python scripts/backtest.py

dashboard:
	streamlit run dashboard/app.py

test:
	pytest tests/ --cov=src

clean:
	find . -type d -name "__pycache__" -exec rm -r {} +
	find . -type f -name "*.pyc" -delete
	rm -rf .pytest_cache
	rm -rf htmlcov
